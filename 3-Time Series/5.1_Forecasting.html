Example 1: Quarterly Australian Beer production

Description

Total quarterly beer production in Australia (in megalitres) from 1956:Q1 to 2008:Q3.

See ?ausbeer for additional information

Plot the Time-series

plot(ausbeer,ylab='Beer',main='Quarterly Australian Beer production')

tsdisplay(ausbeer,main='Monthly Carbon Dioxide Levels at Alert, NWT, Canada')

Split the data into training and test dataset

beer_Train <- window(ausbeer,start=1992,end=2000.75)
beer_Test <- window(ausbeer,start=2001, end=2004)
beer_TrainTest <- window(ausbeer,start=1992,end=2004)

Plot the Training, Test and Combined datasets

win.graph(width=6, height=6,pointsize=12)
par(mfrow=c(3,1))
plot(beer_TrainTest,ylab='Beer',main='Quarterly Australian Beer production [1992 - 2004]')
plot(beer_Train,ylab='Beer',main='Quarterly Australian Beer production [1992 - 2000]')
plot(beer_Test,ylab='Beer',main='Quarterly Australian Beer production [2001 - 2004]')

Define forecast horizon h

h <- 13

Baseline Forecasting

For additional information see: ?meanf - Mean Forecast ?naive - Naive and Random Walk Forecasts ?snaive - Naive and Random Walk Forecasts ?rwf - - Naive and Random Walk Forecasts

Model_Mean <- meanf(beer_Train, h) 
Model_Naive <- naive(beer_Train, h) 
Model_SNaive <- snaive(beer_Train, h)
Model_Drift <- rwf(beer_Train, h, drift=TRUE)

Plot Forecast

autoplot(beer_TrainTest) +
  autolayer(Model_Mean$mean, series="Mean") +
  autolayer(Model_Naive$mean, series="Nave") +
  autolayer(Model_SNaive$mean, series="Seasonal nave") +
  autolayer(Model_Drift$mean, series="Drift") +
  ggtitle("Forecasts for quarterly beer production") +
  xlab("Year") + ylab("Megalitres")
