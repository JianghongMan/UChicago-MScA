CV - Expanding and Sliding Window

Applying time series CV and comparing 1-step, 2-step, ., 12-step forecasts using the Mean Absolute Error (MAE)

k <- 60 # minimum data length for fitting a model
n <- length(a10) # Number of data points

p <- 12 ### Period
H <- 12 # Forecast Horiz
defaultW <- getOption("warn") 
options(warn = -1)

st <- tsp(a10)[1]+(k-2)/p #  gives the start time in time units,

mae_1 <- matrix(NA,n-k,H)
mae_2 <- matrix(NA,n-k,H)


for(i in 1:(n-k))
{
  
  
  ### One Month rolling forecasting
  # Expanding Window 
  train_1 <- window(a10, end=st + i/p)  ## Window Length: k+i
  
  # Sliding Window - keep the training window of fixed length. 
  # The training set always consists of k observations.
  train_2 <- window(a10, start=st+(i-k+1)/p, end=st+i/p) ## Window Length: k
  
  test <- window(a10, start=st + (i+1)/p, end=st + (i+H)/p) ## Window Length: H

  if (i<4) {
  cat(c("*** CV", i,":","len(Expanding Window):",length(train_1), "len(Sliding Window):",length(train_2), "len(Test):",length(test),'\n'  ))
  cat(c("*** TRAIN -  Expanding WIndow:",tsp(train_1)[1],'-',tsp(train_1)[2],'\n'))
  cat(c("*** TRAIN - Sliding WIndow:",tsp(train_2)[1],'-',tsp(train_2)[2],'\n'))
  cat(c("*** TEST:",tsp(test)[1],'-',tsp(test)[2],'\n'))
  cat("*************************** \n \n")
  }
  
  
  fit_1 <- Arima(train_1, order=c(3,0,1), seasonal=list(order=c(0,1,1), period=p),
                include.drift=TRUE, lambda=0, method="ML")
  fcast_1 <- forecast(fit_1, h=H)
  
  
  fit_2 <- Arima(train_2, order=c(3,0,1), seasonal=list(order=c(0,1,1), period=p),
                include.drift=TRUE, lambda=0, method="ML")
  fcast_2 <- forecast(fit_2, h=H)
  
  
  
  mae_1[i,1:length(test)] <- abs(fcast_1[['mean']]-test)
  mae_2[i,1:length(test)] <- abs(fcast_2[['mean']]-test)
  
}

CV - Incremental Window

Another variation which is useful for large data sets isa to use a form of k-fold cross-validation where the training sets increment by several values at a time. For example, instead of incrementing by one observation in each iteration, we could shift the training set forward by 12 observations.

k <- 60 # minimum data length for fitting a model
n <- length(a10) # Number of data points

p <- 12 ### Period
H <- 12 # Forecast Horiz
# because this is based on fewer estimation steps, the results are much more volatile
st <- tsp(a10)[1]+(k-1)/p ### The same as before
mae_3 <- matrix(NA,p,H)

for(i in 1:H)
{
  # Expanding Window 
  train <- window(a10, end=st + (i-1))
  test <- window(a10, start=st + (i-1) + 1/p, end=st + i) ## Window Length: H
  if (i<4) {
    cat(c("*** CV", i,":","len(train):",length(train), "len(Test):",length(test),'\n'  ))
    cat(c("*** TRAIN:",tsp(train)[1],'-',tsp(train)[2],'\n'))
    cat(c("*** TEST:",tsp(test)[1],'-',tsp(test)[2],'\n'))
    cat("*************************** \n \n")
    }
  

  fit_3 <- Arima(train, order=c(3,0,1), seasonal=list(order=c(0,1,1), period=p),
                include.drift=TRUE, lambda=0, method="ML")
  fcast_3 <- forecast(fit_3, h=H)
 
  mae_3[i,1:length(test)] <- abs(fcast_3[['mean']]-test)
}
